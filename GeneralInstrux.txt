 #Add the remote, call it "SourceNCC":

#git remote add upstream https://github.com/whoever/whatever.git
git remote add SourceNCC https://github.com/KaleyatWork/NCC-SHARE.git

# Fetch all the branches of that remote into remote-tracking branches

git fetch SourceNCC

# Make sure that you're on your master branch:

git checkout master

# Rewrite your master branch so that any commits of yours that
# aren't already in SourceNCC/master are replayed on top of that
# other branch:

git rebase SourceNCC/master
